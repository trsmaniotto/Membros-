    <script>
        const WEB_APP_URL = 'https://script.google.com/macros/s/AKfycbwDp2g--eUjynwswAER_MBdKs59mTVsgdcbZcJHSgQsb_pC1lALgv670OT5GsL9roP_UQ/exec'; // SUBSTITUA PELA URL DO SEU APPS SCRIPT
        let membersData = [];
        const courseColumns = ["Fundamento", "Imersão", "Você na Verdade", "Batismo", "Identidade", "Inspiração", "DCH", "NT", "Rendição"];

        document.addEventListener('DOMContentLoaded', () => {
            fetchMembers();
            setupEventListeners();
        });

        async function fetchMembers() {
            try {
                const response = await fetch(WEB_APP_URL);
                membersData = await response.json();
                renderTable(membersData);
            } catch (error) {
                console.error('Erro ao buscar membros:', error);
                alert('Não foi possível carregar os dados dos membros. Verifique a URL do backend.');
            }
        }

        function renderTable(membersToRender) {
            const tbody = document.querySelector('#memberTable tbody');
            tbody.innerHTML = ''; // Limpa a tabela antes de renderizar

            membersToRender.forEach(member => {
                const row = tbody.insertRow();
                row.insertCell().textContent = member.ID;
                row.insertCell().textContent = member.Nome;
                row.insertCell().textContent = member.Grupo;
                
                const phoneCell = row.insertCell();
                if (member.Telefone) {
                    const whatsappLink = document.createElement('a');
                    whatsappLink.href = `https://wa.me/${member.Telefone}`;
                    whatsappLink.target = '_blank';
                    whatsappLink.className = 'whatsapp-btn';
                    whatsappLink.innerHTML = `<img src="https://upload.wikimedia.org/wikipedia/commons/6/6b/WhatsApp.svg" alt="WhatsApp"> ${member.Telefone}`;
                    phoneCell.appendChild(whatsappLink);
                } else {
                    phoneCell.textContent = 'N/A';
                }

                courseColumns.forEach(course => {
                    const cell = row.insertCell();
                    const status = String(member[course]).toLowerCase();
                    const indicator = document.createElement('span');
                    indicator.className = 'status-indicator';
                    if (status === 'sim' || status === 'x') {
                        indicator.classList.add('yes');
                        indicator.title = 'Concluído';
                    } else if (status === 'não' || status === '') {
                        indicator.classList.add('no');
                        indicator.title = 'Não Concluído';
                    } else {
                        indicator.classList.add('unknown');
                        indicator.title = 'Status Desconhecido';
                    }
                    cell.appendChild(indicator);
                });

                const actionsCell = row.insertCell();
                const editBtn = document.createElement('button');
                editBtn.textContent = 'Editar';
                editBtn.className = 'edit-btn';
                editBtn.onclick = () => openMemberModalForEdit(member);
                actionsCell.appendChild(editBtn);

                const deleteBtn = document.createElement('button');
                deleteBtn.textContent = 'Excluir';
                deleteBtn.className = 'delete-btn';
                deleteBtn.onclick = () => deleteMember(member.rowIndex, member.Nome);
                actionsCell.appendChild(deleteBtn);
            });
        }

        function setupEventListeners() {
            document.getElementById('searchInput').addEventListener('keyup', applyFilters);
            document.getElementById('groupFilter').addEventListener('change', applyFilters);
            document.getElementById('courseFilter').addEventListener('change', applyFilters);
            document.getElementById('addMemberBtn').addEventListener('click', openAddMemberModal);
            document.querySelector('.close-button').addEventListener('click', closeMemberModal);
            window.addEventListener('click', (event) => {
                if (event.target == document.getElementById('memberModal')) {
                    closeMemberModal();
                }
            });
            document.getElementById('memberForm').addEventListener('submit', handleMemberFormSubmit);
        }

        function applyFilters() {
            const searchTerm = document.getElementById('searchInput').value.toLowerCase();
            const groupFilter = document.getElementById('groupFilter').value;
            const courseFilter = document.getElementById('courseFilter').value;

            const filteredMembers = membersData.filter(member => {
                const matchesSearch = member.Nome.toLowerCase().includes(searchTerm);
                const matchesGroup = groupFilter === '' || member.Grupo === groupFilter;
                const matchesCourse = courseFilter === '' || 
                                      (courseColumns.includes(courseFilter) && 
                                       (String(member[courseFilter]).toLowerCase() === 'sim' || String(member[courseFilter]).toLowerCase() === 'x'));
                return matchesSearch && matchesGroup && matchesCourse;
            });
            renderTable(filteredMembers);
        }

        function openAddMemberModal() {
            document.getElementById('modalTitle').textContent = 'Adicionar Novo Membro';
            document.getElementById('memberForm').reset();
            document.getElementById('memberId').value = ''; // Limpa o ID para adicionar
            document.getElementById('rowIndex').value = ''; // Limpa o rowIndex
            document.getElementById('memberModal').style.display = 'block';
        }

        function openMemberModalForEdit(member) {
            document.getElementById('modalTitle').textContent = 'Editar Membro';
            document.getElementById('memberForm').reset();
            
            document.getElementById('memberId').value = member.ID;
            document.getElementById('rowIndex').value = member.rowIndex; // Guarda o rowIndex para atualização
            document.getElementById('nome').value = member.Nome;
            document.getElementById('grupo').value = member.Grupo;
            document.getElementById('telefone').value = member.Telefone;

            courseColumns.forEach(course => {
                const checkbox = document.querySelector(`input[name="${course}"]`);
                if (checkbox) {
                    checkbox.checked = (String(member[course]).toLowerCase() === 'sim' || String(member[course]).toLowerCase() === 'x');
                }
            });

            document.getElementById('memberModal').style.display = 'block';
        }

        function closeMemberModal() {
            document.getElementById('memberModal').style.display = 'none';
        }

        async function handleMemberFormSubmit(event) {
            event.preventDefault();
            const form = event.target;
            const formData = new FormData(form);
            const member = {};
            formData.forEach((value, key) => {
                if (courseColumns.includes(key)) {
                    member[key] = value === 'Sim' ? 'Sim' : 'Não'; // Garante 'Sim' ou 'Não' para cursos
                } else {
                    member[key] = value;
                }
            });

            const memberId = document.getElementById('memberId').value;
            const rowIndex = document.getElementById('rowIndex').value;
            let action = memberId ? 'update' : 'add';

            // Se for adicionar, precisamos gerar um novo ID
            if (action === 'add') {
                const maxId = membersData.reduce((max, m) => Math.max(max, parseInt(m.ID || 0)), 0);
                member.ID = (maxId + 1).toString();
            }

            try {
                const response = await fetch(WEB_APP_URL, {
                    method: 'POST',<br/>
                    headers: {<br/>
                        'Content-Type': 'text/plain;charset=utf-8', // Apps Script exige text/plain
                    },
                    body: JSON.stringify({ action: action, member: member, rowIndex: rowIndex }),
                });
                const result = await response.json();
                if (result.success) {
                    alert(result.message);
                    closeMemberModal();
                    fetchMembers(); // Recarrega os dados
                } else {
                    alert('Erro ao salvar membro: ' + result.message);
                }
            } catch (error) {
                console.error('Erro ao salvar membro:', error);
                alert('Erro de comunicação com o servidor ao salvar membro.');
            }
        }

        async function deleteMember(rowIndex, memberName) {
            if (!confirm(`Tem certeza que deseja excluir o membro ${memberName}?`)) {
                return;
            }

            try {
                const response = await fetch(WEB_APP_URL, {
                    method: 'POST',<br/>
                    headers: {<br/>
                        'Content-Type': 'text/plain;charset=utf-8',
                    },
                    body: JSON.stringify({ action: 'delete', rowIndex: rowIndex }),
                });
                const result = await response.json();
                if (result.success) {
                    alert(result.message);
                    fetchMembers(); // Recarrega os dados
                } else {
                    alert('Erro ao excluir membro: ' + result.message);
                }
            } catch (error) {
                console.error('Erro ao excluir membro:', error);
                alert('Erro de comunicação com o servidor ao excluir membro.');
            }
        }
